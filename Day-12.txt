Code:-		
		#include <stdio.h>
		#include <unistd.h>
		int main()
		{
			printf("\nHello World\n");
			int f=fork();//Copeing process two times
			int p=getpid();
			if(f==0)
			{
				printf("\nI'm the child Process %d",p);
			}
			else{
				printf("\nI'm the parent Process %d",f);
			}
			
			return 0;
		}
