9) Get different inputs

Input:-

		int main()
		{
			double number;
			char alpha;

			printf("Enter double input: ");
			scanf("%lf",&number);
			printf("\nYour double input is:%lf",number);

			printf("Enter char input:");
			scanf(" %c",&alpha);

		}
